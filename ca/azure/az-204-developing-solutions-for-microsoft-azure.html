  • Implementant l'Emmagatzematge Blob d'Azure
    • Creant i gestionant l'emmagatzematge Blob
    • Pujant, descarregant i eliminant blobs
using Azure.Storage.Blobs;
using System;
using System.Threading.Tasks;

public class BlobStorageExample
{
    private const string connectionString = "<Your_Connection_String>";
    private const string containerName = "example-container";
    private const string blobName = "example-blob.txt";

    public static async Task Main(string[] args)
    {
        BlobServiceClient blobServiceClient = new BlobServiceClient(connectionString);
        BlobContainerClient containerClient = blobServiceClient.GetBlobContainerClient(containerName);
        await containerClient.CreateIfNotExistsAsync();

        BlobClient blobClient = containerClient.GetBlobClient(blobName);
        await blobClient.UploadAsync("path/to/local/file.txt", true);

        Console.WriteLine("Blob uploaded successfully.");
    }
}

Conclusió

Continua practicant i explorant les àmplies ofertes d'Azure per millorar encara més la teva experiència.
