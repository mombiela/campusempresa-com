HTTP i Servidors Web ►

Les xarxes són un aspecte crucial de moltes aplicacions, ja que permeten la comunicació entre diferents sistemes. Go, amb la seva potent biblioteca estàndard, proporciona un suport robust per a les xarxes. Aquesta secció et guiarà des dels conceptes bàsics fins als avançats en xarxes en Go.

Introducció a les Xarxes en Go

  • Què són les Xarxes?

    • Les xarxes impliquen connectar ordinadors per compartir recursos i informació.
    • Protocols comuns: HTTP, TCP, UDP.

  • Paquets de Xarxes de Go

    • net: Proporciona capacitats bàsiques de xarxes.
    • net/http: Simplifica les implementacions de clients i servidors HTTP.

Conceptes Bàsics de Xarxes

Treballant amb TCP

  • Visió General de TCP

    • Protocol fiable i orientat a la connexió.
    • Assegura que les dades es lliurin en ordre i sense errors.

  • Creant un Servidor TCP

package main

import (
    "fmt"
    "net"
)

func main() {
    listener, err := net.Listen("tcp", ":8080")
    if err != nil {
        fmt.Println("Error creant el listener:", err)
        return
    }
    defer listener.Close()

    fmt.Println("El servidor està escoltant al port 8080...")
    for {
        conn, err := listener.Accept()
        if err != nil {
            fmt.Println("Error acceptant la connexió:", err)
            continue
        }
        go handleConnection(conn)
    }
}

func handleConnection(conn net.Conn) {
    defer conn.Close()
    buffer := make([]byte, 1024)
    for {
        n, err := conn.Read(buffer)
        if err != nil {
            fmt.Println("Error llegint de la connexió:", err)
            return
        }
        fmt.Println("Rebut:", string(buffer[:n]))
        conn.Write([]byte("Missatge rebut"))
    }
}
  • Creant un Client TCP
package main

import (
    "fmt"
    "net"
)

func main() {
    conn, err := net.Dial("tcp", "localhost:8080")
    if err != nil {
        fmt.Println("Error connectant amb el servidor:", err)
        return
    }
    defer conn.Close()

    message := "Hola, Servidor!"
    conn.Write([]byte(message))

    buffer := make([]byte, 1024)
    n, err := conn.Read(buffer)
    if err != nil {
        fmt.Println("Error llegint del servidor:", err)
        return
    }
    fmt.Println("Resposta del servidor:", string(buffer[:n]))
}

Treballant amb UDP

  • Visió General de UDP

    • Sense connexió, menys fiable que TCP.
    • Adequat per a aplicacions on la velocitat és crítica i la pèrdua ocasional de dades és acceptable.

  • Creant un Servidor UDP

package main

import (
    "fmt"
    "net"
)

func main() {
    addr := net.UDPAddr{
        Port: 8080,
        IP:   net.ParseIP("127.0.0.1"),
    }
    conn, err := net.ListenUDP("udp", &addr)
    if err != nil {
        fmt.Println("Error creant el servidor UDP:", err)
        return
    }
    defer conn.Close()

    buffer := make([]byte, 1024)
    for {
        n, clientAddr, err := conn.ReadFromUDP(buffer)
        if err != nil {
            fmt.Println("Error llegint del client UDP:", err)
            continue
        }
        fmt.Println("Rebut:", string(buffer[:n]), "de", clientAddr)
        conn.WriteToUDP([]byte("Missatge rebut"), clientAddr)
    }
}
  • Creant un Client UDP
package main

import (
    "fmt"
    "net"
)

func main() {
    serverAddr := net.UDPAddr{
        Port: 8080,
        IP:   net.ParseIP("127.0.0.1"),
    }
    conn, err := net.DialUDP("udp", nil, &serverAddr)
    if err != nil {
        fmt.Println("Error connectant amb el servidor UDP:", err)
        return
    }
    defer conn.Close()

    message := []byte("Hola, Servidor UDP!")
    conn.Write(message)

    buffer := make([]byte, 1024)
    n, _, err := conn.ReadFromUDP(buffer)
    if err != nil {
        fmt.Println("Error llegint del servidor UDP:", err)
        return
    }
    fmt.Println("Resposta del servidor:", string(buffer[:n]))
}

Xarxes HTTP

Construint un Servidor HTTP

  • Conceptes Bàsics del Servidor HTTP

    • Gestiona sol·licituds i respostes HTTP.
    • Utilitza el paquet net/http.

  • Creant un Servidor HTTP Simple

package main

import (
    "fmt"
    "net/http"
)

func helloHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hola, Món!")
}

func main() {
    http.HandleFunc("/", helloHandler)
    fmt.Println("Iniciant el servidor a :8080")
    http.ListenAndServe(":8080", nil)
}

Construint un Client HTTP

  • Conceptes Bàsics del Client HTTP

    • Envia sol·licituds HTTP i processa respostes.
    • Utilitza el paquet net/http.

  • Creant un Client HTTP Simple

package main

import (
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    resp, err := http.Get("http://localhost:8080")
    if err != nil {
        fmt.Println("Error fent la sol·licitud GET:", err)
        return
    }
    defer resp.Body.Close()

    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        fmt.Println("Error llegint el cos de la resposta:", err)
        return
    }
    fmt.Println("Resposta del servidor:", string(body))
}

Conceptes Avançats de Xarxes

Concurrència en Xarxes

  • Goroutines i Canals

    • Utilitza goroutines per gestionar múltiples connexions de manera concurrent.
    • Canals per a la comunicació segura entre goroutines.

  • Exemple: Servidor TCP Concurrent

package main

import (
    "fmt"
    "net"
    "sync"
)

func main() {
    listener, err := net.Listen("tcp", ":8080")
    if err != nil {
        fmt.Println("Error creant el listener:", err)
        return
    }
    defer listener.Close()

    var wg sync.WaitGroup
    fmt.Println("El servidor està escoltant al port 8080...")
    for {
        conn, err := listener.Accept()
        if err != nil {
            fmt.Println("Error acceptant la connexió:", err)
            continue
        }
        wg.Add(1)
        go func(conn net.Conn) {
            defer wg.Done()
            handleConnection(conn)
        }(conn)
    }
    wg.Wait()
}

func handleConnection(conn net.Conn) {
    defer conn.Close()
    buffer := make([]byte, 1024)
    for {
        n, err := conn.Read(buffer)
        if err != nil {
            fmt.Println("Error llegint de la connexió:", err)
            return
        }
        fmt.Println("Rebut:", string(buffer[:n]))
        conn.Write([]byte("Missatge rebut"))
    }
}

Xarxes Segures amb TLS

  • Visió General de TLS

    • Proporciona encriptació per a una comunicació segura.
    • Utilitza el paquet crypto/tls.

  • Creant un Servidor TLS

package main

import (
    "crypto/tls"
    "fmt"
    "net/http"
)

func helloHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hola, Món Segur!")
}

func main() {
    http.HandleFunc("/", helloHandler)
    server := &http.Server{
        Addr: ":8443",
        TLSConfig: &tls.Config{
            MinVersion: tls.VersionTLS12,
        },
    }
    fmt.Println("Iniciant el servidor segur a :8443")
    server.ListenAndServeTLS("server.crt", "server.key")
}
