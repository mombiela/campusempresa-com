Introducción a los Algoritmos

Los algoritmos son procedimientos o fórmulas paso a paso para resolver problemas. En programación, los algoritmos son esenciales para realizar tareas de manera eficiente y efectiva.

Conceptos Clave

  • Definición: Un conjunto de instrucciones diseñadas para realizar una tarea específica.
  • Características:
    • Entrada: El algoritmo toma cero o más entradas.
    • Salida: El algoritmo produce una o más salidas.
    • Definitud: Cada paso está claramente definido.
    • Finitud: El algoritmo termina después de un número finito de pasos.
    • Efectividad: Cada paso es lo suficientemente básico como para ser llevado a cabo.

Algoritmos Básicos

Algoritmos de Ordenamiento

El ordenamiento es el proceso de organizar datos en un orden particular.

Ordenamiento de Burbuja

El Ordenamiento de Burbuja es un algoritmo de ordenamiento simple basado en comparaciones.

#include <iostream>
using namespace std;

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        for (int j = 0; j < n-i-1; j++) {
            if (arr[j] > arr[j+1]) {
                // Intercambiar arr[j] y arr[j+1]
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr)/sizeof(arr[0]);
    bubbleSort(arr, n);
    cout << "Arreglo ordenado: \n";
    for (int i=0; i < n; i++)
        cout << arr[i] << " ";
    return 0;
}

Ordenamiento por Selección

El Ordenamiento por Selección es otro algoritmo de ordenamiento simple basado en comparaciones.

#include <iostream>
using namespace std;

void selectionSort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        int min_idx = i;
        for (int j = i+1; j < n; j++)
            if (arr[j] < arr[min_idx])
                min_idx = j;
        // Intercambiar el elemento mínimo encontrado con el primer elemento
        int temp = arr[min_idx];
        arr[min_idx] = arr[i];
        arr[i] = temp;
    }
}

int main() {
    int arr[] = {64, 25, 12, 22, 11};
    int n = sizeof(arr)/sizeof(arr[0]);
    selectionSort(arr, n);
    cout << "Arreglo ordenado: \n";
    for (int i=0; i < n; i++)
        cout << arr[i] << " ";
    return 0;
}

Algoritmos de Búsqueda

La búsqueda es el proceso de encontrar un elemento en una estructura de datos.

Búsqueda Lineal

La Búsqueda Lineal es un algoritmo de búsqueda simple que verifica cada elemento hasta encontrar el elemento deseado.

#include <iostream>
using namespace std;

int linearSearch(int arr[], int n, int x) {
    for (int i = 0; i < n; i++)
        if (arr[i] == x)
            return i;
    return -1;
}

int main() {
    int arr[] = {2, 3, 4, 10, 40};
    int x = 10;
    int n = sizeof(arr)/sizeof(arr[0]);
    int result = linearSearch(arr, n, x);
    (result == -1) ? cout << "El elemento no está presente en el arreglo"
                   : cout << "El elemento está presente en el índice " << result;
    return 0;
}

Búsqueda Binaria

La Búsqueda Binaria es un algoritmo de búsqueda más eficiente que funciona en arreglos ordenados.

#include <iostream>
using namespace std;

int binarySearch(int arr[], int l, int r, int x) {
    while (l <= r) {
        int m = l + (r - l) / 2;
        // Verificar si x está presente en mid
        if (arr[m] == x)
            return m;
        // Si x es mayor, ignorar la mitad izquierda
        if (arr[m] < x)
            l = m + 1;
        // Si x es menor, ignorar la mitad derecha
        else
            r = m - 1;
    }
    // Si llegamos aquí, entonces el elemento no estaba presente
    return -1;
}

int main() {
    int arr[] = {2, 3, 4, 10, 40};
    int x = 10;
    int n = sizeof(arr)/sizeof(arr[0]);
    int result = binarySearch(arr, 0, n-1, x);
    (result == -1) ? cout << "El elemento no está presente en el arreglo"
                   : cout << "El elemento está presente en el índice " << result;
    return 0;
}

Algoritmos Avanzados

Programación Dinámica

La Programación Dinámica es un método para resolver problemas complejos dividiéndolos en subproblemas más simples.

Secuencia de Fibonacci

La secuencia de Fibonacci es un ejemplo clásico de programación dinámica.

#include <iostream>
using namespace std;

int fib(int n) {
    int f[n+1];
    f[0] = 0;
    f[1] = 1;
    for (int i = 2; i <= n; i++)
        f[i] = f[i-1] + f[i-2];
    return f[n];
}

int main() {
    int n = 9;
    cout << "El número de Fibonacci es " << fib(n);
    return 0;
}

Algoritmos de Grafos

Los algoritmos de grafos se utilizan para resolver problemas relacionados con la teoría de grafos.

Búsqueda en Profundidad (DFS)

DFS es un algoritmo para recorrer o buscar estructuras de datos de árbol o grafo.

#include <iostream>
#include <list>
using namespace std;

class Graph {
    int V;
    list<int> *adj;
    void DFSUtil(int v, bool visitado[]);
public:
    Graph(int V);
    void addEdge(int v, int w);
    void DFS(int v);
};

Graph::Graph(int V) {
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w) {
    adj[v].push_back(w);
}

void Graph::DFSUtil(int v, bool visitado[]) {
    visitado[v] = true;
    cout << v << " ";
    list<int>::iterator i;
    for (i = adj[v].begin(); i != adj[v].end(); ++i)
        if (!visitado[*i])
            DFSUtil(*i, visitado);
}

void Graph::DFS(int v) {
    bool *visitado = new bool[V];
    for (int i = 0; i < V; i++)
        visitado[i] = false;
    DFSUtil(v, visitado);
}

int main() {
    Graph g(4);
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 2);
    g.addEdge(2, 0);
    g.addEdge(2, 3);
    g.addEdge(3, 3);
    cout << "Recorrido en Profundidad (comenzando desde el vértice 2) \n";
    g.DFS(2);
    return 0;
}
