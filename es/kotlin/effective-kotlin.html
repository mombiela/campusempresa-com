Sintaxis y Características Básicas

  • Variables y Tipos de Datos
    • Inmutable (val) vs Mutable (var)
    val valorInmutable: Int = 10
var valorMutable: Int = 20
  • Tipos de Datos Básicos: Int, Double, String, Boolean
  • Inferencia de Tipos
  • Seguridad Nula 
    var cadenaNullable: String? = "Hola"
cadenaNullable = null

Flujo de Control

  • Declaraciones Condicionales: if, else, when 
    val numero = 10
val resultado = if (numero > 0) "Positivo" else "Negativo"

val dia = 3
val nombreDia = when(dia) {
    1 -> "Lunes"
    2 -> "Martes"
    3 -> "Miércoles"
    else -> "Desconocido"
}
  • Bucles: for, while, do-while 
    for (i in 1..5) {
    println(i)
}

var x = 5
while (x > 0) {
    println(x)
    x--
}

Funciones

  • Definiendo y llamando funciones 
    fun saludar(nombre: String): String {
    return "Hola, $nombre"
}

println(saludar("Kotlin"))
  • Argumentos Predeterminados y Nombrados 
    fun imprimirMensaje(mensaje: String = "Mensaje Predeterminado") {
    println(mensaje)
}

imprimirMensaje()
imprimirMensaje("Mensaje Personalizado")
  • Funciones de una sola expresión 
    fun sumar(a: Int, b: Int) = a + b

Programación Orientada a Objetos

  • Clases y Objetos 
    class Persona(val nombre: String, var edad: Int)

val persona = Persona("Juan", 30)
println(persona.nombre)
  • Herencia 
    open class Animal(val nombre: String)

class Perro(nombre: String) : Animal(nombre)

val perro = Perro("Buddy")
println(perro.nombre)
  • Interfaces 
    interface Conducible {
    fun conducir()
}

class Coche : Conducible {
    override fun conducir() {
        println("Conduciendo un coche")
    }
}

val coche = Coche()
coche.conducir()

Características Avanzadas de Kotlin

  • Clases de Datos 
    data class Usuario(val nombre: String, val edad: Int)

val usuario = Usuario("Alicia", 25)
println(usuario)
  • Clases Selladas 
    sealed class Resultado
data class Exito(val datos: String) : Resultado()
data class Error(val mensaje: String) : Resultado()

fun manejarResultado(resultado: Resultado) {
    when (resultado) {
        is Exito -> println("Éxito: ${resultado.datos}")
        is Error -> println("Error: ${resultado.mensaje}")
    }
}
  • Funciones de Extensión 
    fun String.esPalindromo(): Boolean {
    return this == this.reversed()
}

println("madam".esPalindromo())
  • Corrutinas para programación asíncrona 
    import kotlinx.coroutines.*

fun main() = runBlocking {
    launch {
        delay(1000L)
        println("¡Mundo!")
    }
    println("Hola,")
}

Programación Funcional en Kotlin

  • Lambdas y Funciones de Orden Superior 
    val suma: (Int, Int) -> Int = { a, b -> a + b }
println(suma(1, 2))

fun operar(a: Int, b: Int, operacion: (Int, Int) -> Int): Int {
    return operacion(a, b)
}

println(operar(3, 4, suma))
  • Colecciones y Operaciones Funcionales 
    val numeros = listOf(1, 2, 3, 4, 5)
val duplicados = numeros.map { it * 2 }
println(duplicados)

val filtrados = numeros.filter { it % 2 == 0 }
println(filtrados)

  • Escribir código limpio y legible
  • Usar Kotlin idiomático

    try {
    val resultado = operacionRiesgosa()
    println(resultado)
} catch (e: Exception) {
    println("Ocurrió un error: ${e.message}")
}
