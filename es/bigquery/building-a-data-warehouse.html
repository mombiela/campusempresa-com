Ejemplo: Creación de un Esquema Estrella

-- Crear una tabla de hechos

  sale_id INT64,
  product_id INT64,
  customer_id INT64,
  sale_date DATE,
  sale_amount FLOAT64
);

-- Crear una tabla de dimensión para productos
CREATE TABLE my_dataset.products (
  product_id INT64,
  product_name STRING,
  category STRING,
  price FLOAT64
);

-- Crear una tabla de dimensión para clientes
CREATE TABLE my_dataset.customers (
  customer_id INT64,
  customer_name STRING,
  customer_email STRING,
  customer_address STRING
);

Consulta de Datos en BigQuery

  • Escritura de Consultas SQL en BigQuery
  • Uso de SQL Estándar vs. SQL Legado
  • Mejores Prácticas para la Optimización de Consultas

Ejemplo: Consulta SQL Básica

-- Seleccionar todos los registros de ventas
SELECT * FROM my_dataset.sales;

-- Agregar ventas por producto
SELECT 
  p.product_name,
  SUM(s.sale_amount) AS total_sales
FROM 
  my_dataset.sales s
JOIN 
  my_dataset.products p
ON 
  s.product_id = p.product_id
GROUP BY 
  p.product_name;

Ejemplo: Entrenamiento de un Modelo de Aprendizaje Automático en BigQuery ML

-- Crear un modelo para predecir el monto de ventas
CREATE OR REPLACE MODEL my_dataset.sales_model
OPTIONS(model_type='linear_reg') AS
SELECT
  sale_amount,
  product_id,
  customer_id,
  EXTRACT(DAYOFWEEK FROM sale_date) AS day_of_week
FROM
  my_dataset.sales;
