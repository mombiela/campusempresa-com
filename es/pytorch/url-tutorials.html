Construcción de Redes Neuronales

  • Introducción a las Redes Neuronales
    • ¿Qué son las redes neuronales?
    • Componentes básicos de una red neuronal
  • Uso de torch.nn para Construir Redes
    • Definición de una red neuronal simple
    • Pasos hacia adelante y hacia atrás
import torch.nn as nn
import torch.nn.functional as F

# Definición de una red neuronal simple
class SimpleNN(nn.Module):
    def __init__(self):
        super(SimpleNN, self).__init__()
        self.fc1 = nn.Linear(10, 50)
        self.fc2 = nn.Linear(50, 1)

    def forward(self, x):
        x = F.relu(self.fc1(x))
        x = self.fc2(x)
        return x

# Creación de una instancia de la red
net = SimpleNN()
print(net)

Entrenamiento de Redes Neuronales

  • Preparación de Datos
    • Carga de conjuntos de datos usando torchvision
    • Preprocesamiento y aumento de datos
  • Bucle de Entrenamiento
    • Definición de funciones de pérdida y optimizadores
    • Implementación del bucle de entrenamiento
import torch.optim as optim
from torchvision import datasets, transforms

# Carga de conjuntos de datos
transform = transforms.Compose([transforms.ToTensor()])
train_dataset = datasets.MNIST(root='./data', train=True, download=True, transform=transform)
train_loader = torch.utils.data.DataLoader(dataset=train_dataset, batch_size=64, shuffle=True)

# Definición de función de pérdida y optimizador
criterion = nn.CrossEntropyLoss()
optimizer = optim.SGD(net.parameters(), lr=0.01)

# Bucle de entrenamiento
for epoch in range(2):  # recorrer el conjunto de datos varias veces
    for inputs, labels in train_loader:
        optimizer.zero_grad()  # poner a cero los gradientes de los parámetros
        outputs = net(inputs.view(-1, 10))  # paso hacia adelante
        loss = criterion(outputs, labels)  # calcular la pérdida
        loss.backward()  # paso hacia atrás
        optimizer.step()  # optimizar

print('Entrenamiento Finalizado')
