La red es un aspecto crucial de muchas aplicaciones, permitiendo la comunicación entre diferentes sistemas. Go, con su poderosa biblioteca estándar, proporciona un soporte robusto para redes. Esta sección te guiará desde los conceptos básicos hasta los avanzados en redes en Go.

Introducción a las Redes en Go

  • ¿Qué es la Red?

    • La red implica conectar computadoras para compartir recursos e información.
    • Protocolos comunes: HTTP, TCP, UDP.

  • Paquetes de Redes en Go

    • net: Proporciona capacidades básicas de red.
    • net/http: Simplifica las implementaciones de clientes y servidores HTTP.

Conceptos Básicos de Redes

Trabajando con TCP

  • Visión General de TCP

    • Protocolo confiable y orientado a la conexión.
    • Asegura que los datos se entreguen en orden y sin errores.

  • Creando un Servidor TCP

package main

import (
    "fmt"
    "net"
)

func main() {
    listener, err := net.Listen("tcp", ":8080")
    if err != nil {
        fmt.Println("Error creando el listener:", err)
        return
    }
    defer listener.Close()

    fmt.Println("El servidor está escuchando en el puerto 8080...")
    for {
        conn, err := listener.Accept()
        if err != nil {
            fmt.Println("Error aceptando la conexión:", err)
            continue
        }
        go handleConnection(conn)
    }
}

func handleConnection(conn net.Conn) {
    defer conn.Close()
    buffer := make([]byte, 1024)
    for {
        n, err := conn.Read(buffer)
        if err != nil {
            fmt.Println("Error leyendo de la conexión:", err)
            return
        }
        fmt.Println("Recibido:", string(buffer[:n]))
        conn.Write([]byte("Mensaje recibido"))
    }
}
  • Creando un Cliente TCP
package main

import (
    "fmt"
    "net"
)

func main() {
    conn, err := net.Dial("tcp", "localhost:8080")
    if err != nil {
        fmt.Println("Error conectando al servidor:", err)
        return
    }
    defer conn.Close()

    message := "¡Hola, Servidor!"
    conn.Write([]byte(message))

    buffer := make([]byte, 1024)
    n, err := conn.Read(buffer)
    if err != nil {
        fmt.Println("Error leyendo del servidor:", err)
        return
    }
    fmt.Println("Respuesta del servidor:", string(buffer[:n]))
}

Trabajando con UDP

  • Visión General de UDP

    • Sin conexión, menos confiable que TCP.
    • Adecuado para aplicaciones donde la velocidad es crítica y la pérdida ocasional de datos es aceptable.

  • Creando un Servidor UDP

package main

import (
    "fmt"
    "net"
)

func main() {
    addr := net.UDPAddr{
        Port: 8080,
        IP:   net.ParseIP("127.0.0.1"),
    }
    conn, err := net.ListenUDP("udp", &addr)
    if err != nil {
        fmt.Println("Error creando el servidor UDP:", err)
        return
    }
    defer conn.Close()

    buffer := make([]byte, 1024)
    for {
        n, clientAddr, err := conn.ReadFromUDP(buffer)
        if err != nil {
            fmt.Println("Error leyendo del cliente UDP:", err)
            continue
        }
        fmt.Println("Recibido:", string(buffer[:n]), "de", clientAddr)
        conn.WriteToUDP([]byte("Mensaje recibido"), clientAddr)
    }
}
  • Creando un Cliente UDP
package main

import (
    "fmt"
    "net"
)

func main() {
    serverAddr := net.UDPAddr{
        Port: 8080,
        IP:   net.ParseIP("127.0.0.1"),
    }
    conn, err := net.DialUDP("udp", nil, &serverAddr)
    if err != nil {
        fmt.Println("Error conectando al servidor UDP:", err)
        return
    }
    defer conn.Close()

    message := []byte("¡Hola, Servidor UDP!")
    conn.Write(message)

    buffer := make([]byte, 1024)
    n, _, err := conn.ReadFromUDP(buffer)
    if err != nil {
        fmt.Println("Error leyendo del servidor UDP:", err)
        return
    }
    fmt.Println("Respuesta del servidor:", string(buffer[:n]))
}

Redes HTTP

Construyendo un Servidor HTTP

  • Conceptos Básicos del Servidor HTTP

    • Maneja solicitudes y respuestas HTTP.
    • Utiliza el paquete net/http.

  • Creando un Servidor HTTP Simple

package main

import (
    "fmt"
    "net/http"
)

func helloHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "¡Hola, Mundo!")
}

func main() {
    http.HandleFunc("/", helloHandler)
    fmt.Println("Iniciando el servidor en :8080")
    http.ListenAndServe(":8080", nil)
}

Construyendo un Cliente HTTP

  • Conceptos Básicos del Cliente HTTP

    • Envía solicitudes HTTP y procesa respuestas.
    • Utiliza el paquete net/http.

  • Creando un Cliente HTTP Simple

package main

import (
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    resp, err := http.Get("http://localhost:8080")
    if err != nil {
        fmt.Println("Error realizando la solicitud GET:", err)
        return
    }
    defer resp.Body.Close()

    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        fmt.Println("Error leyendo el cuerpo de la respuesta:", err)
        return
    }
    fmt.Println("Respuesta del servidor:", string(body))
}

Conceptos Avanzados de Redes

Concurrencia en Redes

  • Goroutines y Canales

    • Usa goroutines para manejar múltiples conexiones concurrentemente.
    • Canales para comunicación segura entre goroutines.

  • Ejemplo: Servidor TCP Concurrente

package main

import (
    "fmt"
    "net"
    "sync"
)

func main() {
    listener, err := net.Listen("tcp", ":8080")
    if err != nil {
        fmt.Println("Error creando el listener:", err)
        return
    }
    defer listener.Close()

    var wg sync.WaitGroup
    fmt.Println("El servidor está escuchando en el puerto 8080...")
    for {
        conn, err := listener.Accept()
        if err != nil {
            fmt.Println("Error aceptando la conexión:", err)
            continue
        }
        wg.Add(1)
        go func(conn net.Conn) {
            defer wg.Done()
            handleConnection(conn)
        }(conn)
    }
    wg.Wait()
}

func handleConnection(conn net.Conn) {
    defer conn.Close()
    buffer := make([]byte, 1024)
    for {
        n, err := conn.Read(buffer)
        if err != nil {
            fmt.Println("Error leyendo de la conexión:", err)
            return
        }
        fmt.Println("Recibido:", string(buffer[:n]))
        conn.Write([]byte("Mensaje recibido"))
    }
}

Redes Seguras con TLS

  • Visión General de TLS

    • Proporciona encriptación para comunicación segura.
    • Utiliza el paquete crypto/tls.

  • Creando un Servidor TLS

package main

import (
    "crypto/tls"
    "fmt"
    "net/http"
)

func helloHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "¡Hola, Mundo Seguro!")
}

func main() {
    http.HandleFunc("/", helloHandler)
    server := &http.Server{
        Addr: ":8443",
        TLSConfig: &tls.Config{
            MinVersion: tls.VersionTLS12,
        },
    }
    fmt.Println("Iniciando el servidor seguro en :8443")
    server.ListenAndServeTLS("server.crt", "server.key")
}
